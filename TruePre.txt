    y ~ dbin(ap, n)
    y.pre ~ dbin(ap, m)
    ap<-main.ap*main.Se+(1-main.ap)*(1-main.Sp)

    main.ap  ~ dbeta(a,b)

    main.Sp ~ dbeta(asp, bsp)
